# Total tract area (sqm → acres; CRS 26912 is metric)
total_acres_df <- data.frame(
  GEOID = oz_tracts$GEOID,
  total_acres = as.numeric(sf::st_area(oz_tracts)) / 4046.8564
)

# OZ 1.0 overlap — oz_1 already filtered to state at read time
oz1_overlap <- suppressWarnings(
  sf::st_intersection(
    oz_tracts |> dplyr::select(GEOID),
    oz_1 |> dplyr::select(OBJECTID)
  )
) |>
  dplyr::mutate(overlap_m2 = as.numeric(sf::st_area(geom))) |>
  sf::st_drop_geometry() |>
  dplyr::filter(overlap_m2 > 0) |>
  dplyr::group_by(GEOID) |>
  dplyr::summarize(oz1_acres = sum(overlap_m2) / 4046.8564, .groups = "drop")

# Developable acres — area-proportioned from ATO DEVACRES where TAZ boundaries
# cross tract boundaries: contribution = DEVACRES × (intersection_area / TAZ_area)
ato_areas_df <- data.frame(
  TAZID = ato$TAZID,
  taz_area = as.numeric(sf::st_area(ato))
)
dev_acres_df <- suppressWarnings(
  sf::st_intersection(
    oz_tracts |> dplyr::select(GEOID),
    ato |> dplyr::select(TAZID, DEVACRES)
  )
) |>
  dplyr::mutate(int_area = as.numeric(sf::st_area(geom))) |>
  sf::st_drop_geometry() |>
  dplyr::filter(int_area > 0) |>
  dplyr::left_join(ato_areas_df, by = "TAZID") |>
  dplyr::mutate(dev_acres_contrib = DEVACRES * (int_area / taz_area)) |>
  dplyr::group_by(GEOID) |>
  dplyr::summarize(dev_acres = sum(dev_acres_contrib), .groups = "drop")

# Assemble: one row per OZ 2.0 candidate tract, sorted by county then tract ID
tract_info <- oz_tracts |>

  ) |>
  dplyr::left_join(dev_acres_df, by = "GEOID") |>
  dplyr::arrange(county, GEOID)

# Assign each housing polygon to the tract containing its centroid.
# Centroid assignment is appropriate for parcel-level inventory data — units
# belong to one tract; area-proportioning is not needed.
housing_joined <- sf::st_join(
  sf::st_centroid(housing_inv),
  dplyr::select(oz_tracts, GEOID),
  join = sf::st_within
) |>
  sf::st_drop_geometry() |>
  dplyr::filter(!is.na(GEOID))

# Totals per tract
housing_totals <- housing_joined |>
  dplyr::group_by(GEOID) |>
  dplyr::summarize(
    units_total = sum(UNIT_COUNT, na.rm = TRUE),
    res_acres_total = sum(ACRES, na.rm = TRUE),
    .groups = "drop"
  )

# Unit counts by TYPE — pivot wide; column names reflect actual TYPE values
# in the data, so verify TYPE values before interpreting columns
housing_by_type <- housing_joined |>
  dplyr::group_by(GEOID, TYPE) |>
  dplyr::summarize(units = sum(UNIT_COUNT, na.rm = TRUE), .groups = "drop") |>
  tidyr::pivot_wider(
    names_from = TYPE,
    values_from = units,
    values_fill = 0L,
    names_prefix = "units_"
  )

housing_acres <- housing_joined |>
  dplyr::mutate(
    acre_cat = dplyr::case_when(
      TYPE == "single_family" ~ "sfd",
      TYPE == "multi_family"  ~ "mf_sfa",
      TRUE ~ NA_character_
    )
  ) |>
  dplyr::filter(!is.na(acre_cat)) |>
  dplyr::group_by(GEOID, acre_cat) |>
  dplyr::summarize(acres = sum(ACRES, na.rm = TRUE), .groups = "drop") |>
  tidyr::complete(GEOID, acre_cat = c("sfd", "mf_sfa"), fill = list(acres = 0)) |>
  tidyr::pivot_wider(
    names_from = acre_cat,
    values_from = acres,
    values_fill = 0,
    names_prefix = "res_acres_"
  )

housing_info <- oz_tracts |>
  sf::st_drop_geometry() |>
  dplyr::select(GEOID) |>
  dplyr::left_join(housing_totals, by = "GEOID") |>

# Distance to nearest stop — computed for all tracts via spatial index;
# set to NA for tracts that already have stops within
nearest_stop_idx <- sf::st_nearest_feature(oz_tracts, uta_stops)
transit_dist_df <- data.frame(
  GEOID = oz_tracts$GEOID,
  dist_edge_mi = as.numeric(sf::st_distance(
    oz_tracts,
    uta_stops[nearest_stop_idx, ],
    by_element = TRUE
  )) *
    0.000621371,
  dist_centroid_mi = as.numeric(sf::st_distance(
    suppressWarnings(sf::st_centroid(oz_tracts)),
    uta_stops[nearest_stop_idx, ],
    by_element = TRUE
  )) *
    0.000621371
)

# ---- ATO transit scores — weighted avg by proportioned DEVACRES ----

# 2) Composite ATO score — "both modes both ways":
#    min-max normalize each of the 4 components to 0–100 across all tracts,
#    then average. Yields a single 0–100 index for cross-tract comparison.
ato_composite_df <- transit_info |>
  dplyr::select(GEOID, ato_jobtransit, ato_hhtransit) |>
  dplyr::left_join(
    dplyr::select(auto_info, GEOID, ato_jobauto, ato_hhauto),
    by = "GEOID"
  ) |>
  dplyr::mutate(
    dplyr::across(
      c(ato_jobtransit, ato_hhtransit, ato_jobauto, ato_hhauto),
      ~ {
        mn <- min(.x, na.rm = TRUE)
        mx <- max(.x, na.rm = TRUE)
        if (mx > mn) (.x - mn) / (mx - mn) * 100 else 0
      },
      .names = "{.col}_norm"
    ),
    ato_composite = rowMeans(
      dplyr::pick(dplyr::ends_with("_norm")),
      na.rm = TRUE
    )
  ) |>
  dplyr::select(GEOID, ato_composite)

# Helper: area-proportion (YEAR2037 - YEAR2027) from a TAZ projection layer
# onto oz_tracts. Uses plain area proportioning — no DEVACRES in these layers.
.proportion_growth <- function(taz_layer, value_col) {
  taz_areas <- data.frame(
    TAZID = taz_layer$TAZID,
    taz_area = as.numeric(sf::st_area(taz_layer))
  )
  suppressWarnings(
    sf::st_intersection(
      dplyr::select(oz_tracts, GEOID),
      dplyr::select(taz_layer, TAZID, YEAR2027, YEAR2037)
    )
  ) |>
    dplyr::mutate(int_area = as.numeric(sf::st_area(geom))) |>
    sf::st_drop_geometry() |>

# Expand bbox by a scale factor from its center, preserving CRS class/attributes
.expand_bbox <- function(bbox, factor) {
  cx <- (bbox["xmin"] + bbox["xmax"]) / 2
  cy <- (bbox["ymin"] + bbox["ymax"]) / 2
  hw <- (bbox["xmax"] - bbox["xmin"]) / 2 * factor
  hh <- (bbox["ymax"] - bbox["ymin"]) / 2 * factor
  bbox["xmin"] <- cx - hw
  bbox["xmax"] <- cx + hw
  bbox["ymin"] <- cy - hh
  bbox["ymax"] <- cy + hh
  bbox
}

# Pad bbox so its width:height matches the target figure ratio, preventing dead
# space when coord_sf enforces the geographic aspect ratio of the extent.
.enforce_aspect <- function(bbox, width, height) {
  target <- width / height
  bw <- bbox["xmax"] - bbox["xmin"]
  bh <- bbox["ymax"] - bbox["ymin"]
  cx <- (bbox["xmin"] + bbox["xmax"]) / 2
  cy <- (bbox["ymin"] + bbox["ymax"]) / 2
  if ((bw / bh) > target) {
    half <- bw / target / 2
    bbox["ymin"] <- cy - half
    bbox["ymax"] <- cy + half
  } else {
    half <- bh * target / 2
    bbox["xmin"] <- cx - half
    bbox["xmax"] <- cx + half
  }
  bbox
}

make_tract_map <- function(geoid) {
  focal <- dplyr::filter(oz_tracts, GEOID == geoid)
  other_oz <- dplyr::filter(oz_tracts, GEOID != geoid)
  exp_bb <- .enforce_aspect(
    .expand_bbox(sf::st_bbox(focal), factor = 1.5),
    map_width_in,
    map_height_in
  )
  view <- sf::st_set_crs(sf::st_as_sfc(exp_bb), sf::st_crs(crs))

  ggplot2::ggplot() +
    # 1. Background census tracts — warm light gray, very recessive
    ggplot2::geom_sf(
      data = sf::st_filter(tracts, view),
      fill = "#EDE8E1",
      color = "#D8D2C8",
      linewidth = 0.1
    ) +
    # 2. OZ 1.0 tracts — warm rose-gray underlay shows prior designation
    ggplot2::geom_sf(
      data = sf::st_filter(oz_1, view),
      fill = "#D4C8C4",
      color = NA,
      alpha = 0.5
    ) +
    # 3. Wasatch Choice centers and districts
    ggplot2::geom_sf(
      data = sf::st_filter(wc_centers, view),
      ggplot2::aes(fill = CenterType),
      color = NA,
      alpha = 0.75
    ) +
    ggplot2::scale_fill_manual(
      values = wc_center_colors,
      name = NULL,
      na.value = "#C8C0C0",
      drop = TRUE
    ) +
    # 4. Other OZ 2.0 candidate tracts — pale straw with amber border
    ggplot2::geom_sf(
      data = sf::st_filter(other_oz, view),
      fill = "#F2E8C4",
      color = "#C4A43A",
      linewidth = 0.25,
      alpha = 0.5
    ) +
    # 5. SAP buffers — cool blue over warm palette; temperature contrast reads clearly
    ggplot2::geom_sf(
      data = sf::st_filter(sap_buffers, view),
      fill = "#B8D4E8",
      color = "#6FA4C4",
      linewidth = 0.25,
      alpha = 0.45
    ) +
    # 6. County boundary — thin charcoal, crisp orientation anchor
    ggplot2::geom_sf(
      data = sf::st_filter(counties, view),
      fill = NA,
      color = "#5A5A5A",
      linewidth = 0.35
    ) +
    # 7. Freeway interchanges — solid dark-slate triangle; no white halo at this size
    ggplot2::geom_sf(
      data = sf::st_filter(freeway_exits, view),
      shape = 17,
      color = "#3A4A5A",
      size = 2.0
    ) +
    # 8. UTA Rail / BRT stops — deep burgundy; white stroke separates from dark fills
    ggplot2::geom_sf(
      data = sf::st_filter(uta_stops, view),
      shape = 21,
      fill = "#943030",
      color = "white",
      size = 2.6,
      stroke = 0.4
    ) +
    # 9. Focal tract — opaque charcoal border; fill alpha via scales::alpha() so
    #    only the fill is translucent, leaving the stroke at full opacity
    ggplot2::geom_sf(
      data = focal,
      fill = scales::alpha("#FFFFF0", 0.25),
      color = "#1A1A1A",
      linewidth = 1.2
    ) +
    ggplot2::coord_sf(
      xlim = c(exp_bb["xmin"], exp_bb["xmax"]),
      ylim = c(exp_bb["ymin"], exp_bb["ymax"]),
      crs = sf::st_crs(crs),
      default_crs = sf::st_crs(crs),
      datum = sf::st_crs(crs),
      expand = FALSE
    ) +
    ggplot2::theme_void() +
    ggplot2::theme(
      legend.position = "none",
      plot.background = ggplot2::element_rect(fill = map_bg, color = NA),
      plot.margin = ggplot2::margin(0, 0, 0, 0)
    )
}
